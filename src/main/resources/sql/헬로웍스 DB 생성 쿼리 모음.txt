	EMP_NO		NUMBER	PRIMARY KEY,
	EMP_PWD	VARCHAR2(200)	NOT NULL,	
	EMP_NAME	VARCHAR2(100)	NOT NULL,	
	EMP_EN		VARCHAR2(100)	NOT NULL,	
	EMP_PID		VARCHAR2(100)	UNIQUE NOT NULL,

--- 전자결재
CREATE TABLE APPROVAL (
	APPROVAL_NO	    NUMBER			    PRIMARY KEY,
	APPROVAL_CLASS	VARCHAR2(50)		NOT NULL,
	DETAIL_CLASS	VARCHAR2(50)		NOT NULL,
	TITLE		    VARCHAR2(70)		NOT NULL,
	WRITER		    NUMBER			    NOT NULL,
    WRITER_JOB      VARCHAR2(50)        NOT NULL,
	CREATE_DATE	    DATE			    DEFAULT SYSDATE NOT NULL, 
	DEPT_DNAME	    VARCHAR2(30)		NOT NULL,
    AP_CONTENT	    VARCHAR2(4000)		NOT NULL,
	COOPERATION	    VARCHAR2(20),
    COO_NAME        VARCHAR2(100),
    COO_JOB         VARCHAR2(30),
	DEPT_SHARE	    VARCHAR2(1)		DEFAULT 'N' NOT NULL,
	PROGRESS	    VARCHAR2(20)	DEFAULT '진행중'	NOT NULL,	
	STATUS		    VARCHAR2(1)		NOT NULL,
	FOREIGN KEY (WRITER) REFERENCES EMPLOYEE (EMP_NO),
	FOREIGN KEY (DEPT_DNAME) REFERENCES DEPT (DEPT_DNAME),
	CHECK (DEPT_SHARE IN ('Y', 'N')),
	CHECK (PROGRESS IN ( '임시저장','진행중', '반려', '전결', '결재완료')),
	CHECK (APPROVAL_CLASS IN ('일반', '지출', '인사', '근태')),
	CHECK (DETAIL_CLASS IN('기안', '공문', '회의', '지출', '인사', '근태')),
	CHECK(STATUS IN('Y', 'N'))	    
);

COMMENT ON COLUMN APPROVAL.APPROVAL_NO IS '문서번호';
COMMENT ON COLUMN APPROVAL.APPROVAL_CLASS IS '결재분류';
COMMENT ON COLUMN APPROVAL.DETAIL_CLASS IS '세부분류';
COMMENT ON COLUMN APPROVAL.TITLE IS '제목';
COMMENT ON COLUMN APPROVAL.WRITER IS '작성자';
COMMENT ON COLUMN APPROVAL.WRITER_JOB IS '작성자직급';
COMMENT ON COLUMN APPROVAL.CREATE_DATE IS '작성일자';
COMMENT ON COLUMN APPROVAL.DEPT_DNAME IS '하위부서이름';
COMMENT ON COLUMN APPROVAL.AP_CONTENT IS '결재내용';
COMMENT ON COLUMN APPROVAL.COOPERATION IS '협조';
COMMENT ON COLUMN APPROVAL.COO_NAME IS '협조자이름';
COMMENT ON COLUMN APPROVAL.COO_JOB IS '협조자직급';
COMMENT ON COLUMN APPROVAL.DEPT_SHARE IS '부서공유';
COMMENT ON COLUMN APPROVAL.PROGRESS IS '진행상태';
COMMENT ON COLUMN APPROVAL.STATUS IS '문서상태';

CREATE SEQUENCE SEQ_EA_AP_NO
	START WITH 1
	INCREMENT BY 1
	MINVALUE 1
	MAXVALUE 99999
	NOCACHE
	;
